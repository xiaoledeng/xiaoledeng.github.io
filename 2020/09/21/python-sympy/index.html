2.求导的基本使用

from sympy import *

x,y,z = symbols('x y z')
expr=sin(x)*exp(x)*z

diff_expr=diff(expr, x)
diff_expr2=diff(expr,x,2)

print(diff_expr)
print(diff_expr2)
