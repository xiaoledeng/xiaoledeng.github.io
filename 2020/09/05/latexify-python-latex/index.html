import math 		# еҜје…ҘmathжЁЎеқ—
import latexify 	# еҜје…ҘlatexifyжЁЎеқ—

# зӨәдҫӢдёҖ
# е®ҡд№үеҮҪж•°
@latexify.with_latex
def solve(a, b, c):
  return (-b + math.sqrt(b**2 - 4*a*c)) / (2*a)

print(solve(1, 4, 3)) 	# и®Ўз®—еҮҪж•°еҖј
print(solve)			# еҫ—еҲ°еҮҪж•°зҡ„LaTeXе…¬ејҸд»Јз Ғ
solve					# еңЁjupyter notebookдёӯзӣҙжҺҘеұ•зӨәеҮҪж•°е…¬ејҸ

# зӨәдҫӢдәҢ
# е®ҡд№үеҮҪж•°
@latexify.with_latex
def sinc(x):
  if x == 0:
    return 1
  else:
    return math.sin(x) / x

print(sinc)				# еҫ—еҲ°еҮҪж•°зҡ„LaTeXе…¬ејҸд»Јз Ғ
sinc					# еңЁjupyter notebookдёӯзӣҙжҺҘеұ•зӨәеҮҪж•°е…¬ејҸ

# зӨәдҫӢдёү
# е®ҡд№үеҮҪж•°
# Elif or nested else-if are unrolled.
@latexify.with_latex
def fib(x):
  if x == 0:
    return 1
  elif x == 1:
    return 1
  else:
    return fib(x-1) + fib(x-2)

print(fib)				# еҫ—еҲ°еҮҪж•°зҡ„LaTeXе…¬ејҸд»Јз Ғ
fib						# еңЁjupyter notebookдёӯзӣҙжҺҘеұ•зӨәеҮҪж•°е…¬ејҸ

# зӨәдҫӢеӣӣ
# Some math symbols are converted automatically.
@latexify.with_latex
def greek(alpha, beta, gamma, Omega):
  return alpha * beta + math.gamma(gamma) + Omega

print(greek)			# еҫ—еҲ°еҮҪж•°зҡ„LaTeXе…¬ејҸд»Јз Ғ
greek					# еңЁjupyter notebookдёӯзӣҙжҺҘеұ•зӨәеҮҪж•°е…¬ејҸ
