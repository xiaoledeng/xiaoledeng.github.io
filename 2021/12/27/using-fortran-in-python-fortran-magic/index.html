2. Usage

In jupyter notebook:

In[1]: %load_ext fortranmagic

In[2]: %%fortran
        subroutine f1(x, y, z)
        real, intent(in) :: x,y
        real, intent(out) :: z
        z = sin(x+y)
        end subroutine f1

In[3]: f1(1.0, 2.1415)
Out[3]: 9.26574066397734e-05

In[4]: %%fortran
        subroutine my_function(x, y, z)
            real, intent(in) :: x(:), y(:)
            real, intent(out) :: z(size(x))
            ! using vector operations
            z(:) = sin(x(:) + y(:))
        end subroutine

In[5]: 
import numpy
x = numpy.random.normal(size=10)
y = numpy.random.normal(size=10)
z = my_function(x, y)
z
