macOS对图片批量重命名

2024-11-06
#Unix #Shell #Automation #macOS 
count=1     # 初始化计数器，将变量 count 设置为 1

for file in *.PNG; do       # 使用 for 循环遍历当前文件夹下所有的 .PNG 文件
  if [ -f "$file" ]; then   # 检查 file 是否为一个普通文件，以避免误操作
    new_name=$(printf "image%02d.png" "$count")     # 使用 printf 格式化 count 为两位数
    mv "$file" "$new_name"  # 将当前文件 file 重命名为 new_name
    ((count++))             # 将 count 增加 1
  fi
done
