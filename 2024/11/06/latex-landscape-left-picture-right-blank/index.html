LaTeX横置页面并图片占左侧、右侧空白

2024-11-06
#LaTeX 
\documentclass{article}
\usepackage{graphicx}
\usepackage{geometry}
\usepackage{lscape} % 用于横置页面

\geometry{a4paper, margin=1in}
\pagenumbering{gobble} % 去掉页码

\begin{document}

\begin{landscape} % 使整个页面横向
    \begin{figure}[h!]
        \centering
        % 设置图片在左边部分的中间
        \hspace*{-15cm} % 向左偏移以调整位置，可根据需要调整偏移距离
        \begin{minipage}[t]{0.7\textwidth}
            % 插入图片
            \includegraphics[width=\linewidth]{001.png} % 替换为你的图片文件名
        \end{minipage}
    \end{figure}
\end{landscape}

\end{document}
